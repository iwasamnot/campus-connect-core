rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is admin
    function isAdmin() {
      return request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role != null &&
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin1');
    }
    
    // Messages collection
    match /messages/{messageId} {
      // Allow authenticated users to read all messages
      allow read: if request.auth != null;
      
      // Allow authenticated users to create messages (with attachments, replies, mentions)
      allow create: if request.auth != null &&
        request.resource.data.userId == request.auth.uid;
      
      // Allow users to update their own messages (for editing)
      // Allow any authenticated user to update readBy field (read receipts)
      // Allow any authenticated user to update reactions field
      // Allow admins to pin/unpin messages
      // Allow admins to update any field
      allow update: if request.auth != null && (
        (resource.data.userId == request.auth.uid && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'toxicityConfidence', 'toxicityReason', 'toxicityMethod', 'edited', 'editedAt', 'reactions', 'attachment', 'replyTo', 'replyToText', 'replyToUserName', 'mentions'])) ||
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy'])) ||
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reactions'])) ||
        (isAdmin() && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['pinned', 'pinnedAt'])) ||
        isAdmin()
      );
      
      // Allow message authors and admins to delete messages
      allow delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        isAdmin()
      );
    }
    
    // Users collection
    match /users/{userId} {
      // Allow authenticated users to read any user document
      allow read: if request.auth != null;
      
      // Allow users to create their own user document (during registration)
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Allow users to update their own user document
      // Students can update their profile fields: name, studentEmail, personalEmail, phoneNumber, bio, profilePicture, course, yearOfStudy, dateOfBirth, address, updatedAt, lastSeen, isOnline
      // Admins can update any user document (including other users)
      allow update: if request.auth != null && (
        isAdmin() ||
        (request.auth.uid == userId && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly([
           'name', 
           'studentEmail', 
           'personalEmail', 
           'phoneNumber', 
           'bio', 
           'profilePicture', 
           'course', 
           'yearOfStudy', 
           'dateOfBirth', 
           'address', 
           'updatedAt', 
           'lastSeen', 
           'isOnline'
         ]))
      );
      
      // Allow admins to delete users
      allow delete: if isAdmin();
    }
    
    // Group messages collection (top-level, if used)
    match /groupMessages/{messageId} {
      // Allow authenticated users to read group messages
      allow read: if request.auth != null;
      
      // Allow authenticated users to create group messages
      allow create: if request.auth != null &&
        request.resource.data.userId == request.auth.uid;
      
      // Allow users to update their own messages (for editing)
      // Allow any authenticated user to update readBy field (read receipts)
      // Allow any authenticated user to update reactions field
      // Allow admins to update any field
      allow update: if request.auth != null && (
        (resource.data.userId == request.auth.uid && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'toxicityConfidence', 'toxicityReason', 'toxicityMethod', 'edited', 'editedAt', 'reactions', 'attachment', 'replyTo', 'replyToText', 'replyToUserName', 'mentions'])) ||
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy'])) ||
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reactions'])) ||
        isAdmin()
      );
      
      // Allow message authors and admins to delete messages
      allow delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        isAdmin()
      );
    }
    
    // Reports collection
    match /reports/{reportId} {
      // Allow authenticated users to create reports
      allow create: if request.auth != null;
      
      // Allow admins to read all reports
      allow read: if isAdmin();
      
      // Allow admins to update report status
      allow update: if isAdmin();
      
      // Allow admins to delete reports
      allow delete: if isAdmin();
    }
    
    // Audit logs collection
    match /auditLogs/{logId} {
      // Only admins can read audit logs
      allow read: if isAdmin();
      
      // Only admins can create audit logs (via backend/admin functions)
      allow create: if isAdmin();
      
      // Only admins can update audit logs
      allow update: if isAdmin();
      
      // Only admins can delete audit logs
      allow delete: if isAdmin();
    }
    
    // Groups collection
    match /groups/{groupId} {
      // Allow authenticated users to read all groups (for browsing)
      allow read: if request.auth != null;
      
      // Allow authenticated users to create groups
      allow create: if request.auth != null && 
        request.resource.data.createdBy == request.auth.uid &&
        request.resource.data.members != null &&
        request.auth.uid in request.resource.data.members &&
        request.resource.data.admins != null &&
        request.auth.uid in request.resource.data.admins; // Creator must be admin
      
      // Allow group members to update group (add/remove members, update name/description)
      allow update: if request.auth != null && (
        resource.data.members != null && request.auth.uid in resource.data.members &&
        request.resource.data.diff(resource.data).affectedKeys().hasAny(['name', 'description', 'updatedAt'])
      );
      
      // Allow group admins to update members (add/remove members, approve/deny requests)
      allow update: if request.auth != null && (
        resource.data.admins != null && request.auth.uid in resource.data.admins &&
        request.resource.data.diff(resource.data).affectedKeys().hasAny(['members', 'admins', 'joinRequests', 'updatedAt'])
      );
      
      // Allow any authenticated user to request to join (add themselves to joinRequests)
      allow update: if request.auth != null && 
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['joinRequests', 'updatedAt']) &&
        request.resource.data.joinRequests != null &&
        request.auth.uid in request.resource.data.joinRequests;
      
      // Allow group admins to delete groups
      allow delete: if request.auth != null && (
        resource.data.admins != null && request.auth.uid in resource.data.admins
      );
      
      // Join requests subcollection (if used)
      match /joinRequests/{requestId} {
        // Allow group admins to read join requests
        allow read: if request.auth != null && 
          exists(/databases/$(database)/documents/groups/$(groupId)) &&
          get(/databases/$(database)/documents/groups/$(groupId)).data.admins != null &&
          request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.admins;
        
        // Allow users to create their own join request
        allow create: if request.auth != null && 
          request.resource.data.userId == request.auth.uid;
        
        // Allow group admins to update/delete join requests (approve/deny)
        allow update, delete: if request.auth != null && 
          exists(/databases/$(database)/documents/groups/$(groupId)) &&
          get(/databases/$(database)/documents/groups/$(groupId)).data.admins != null &&
          request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.admins;
      }
    }
    
    // Group messages collection (subcollection)
    match /groups/{groupId}/groupMessages/{messageId} {
      // Allow group members to read messages
      allow read: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members;
      
      // Allow group members to create messages
      allow create: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members &&
        request.resource.data.userId == request.auth.uid;
      
      // Allow message authors to update their own messages (for editing)
      // Allow any group member to update readBy field (read receipts)
      // Allow any group member to update reactions field
      allow update: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members && (
          (resource.data.userId == request.auth.uid && 
           request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'toxicityConfidence', 'toxicityReason', 'toxicityMethod', 'edited', 'editedAt', 'reactions', 'attachment', 'replyTo', 'replyToText', 'replyToUserName', 'mentions'])) ||
          (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy'])) ||
          (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reactions']))
        );
      
      // Allow message authors and group admins to delete messages
      allow delete: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members && (
          resource.data.userId == request.auth.uid ||
          (get(/databases/$(database)/documents/groups/$(groupId)).data.admins != null &&
           request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.admins)
        );
    }
    
    // Private chats collection
    match /privateChats/{chatId} {
      // Allow users to read/create chats they are participants in
      // Allow reading non-existent documents (for checking if chat exists) or if user is participant
      // When document exists, resource.data is available; when it doesn't, we allow read for authenticated users
      allow read: if request.auth != null && (
        !exists(/databases/$(database)/documents/privateChats/$(chatId)) ||
        (resource.data.participants != null &&
         request.auth.uid in resource.data.participants)
      );
      
      allow create: if request.auth != null && 
        request.resource.data.participants != null &&
        request.auth.uid in request.resource.data.participants;
      
      // Allow participants to update chat metadata (lastMessage, lastMessageTime, disappearing messages settings)
      allow update: if request.auth != null && 
        resource.data.participants != null &&
        request.auth.uid in resource.data.participants &&
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['lastMessage', 'lastMessageTime', 'disappearingMessagesEnabled', 'disappearingMessagesDuration']);
      
      // Private chat messages subcollection
      match /messages/{messageId} {
        // Helper function to check if user is participant
        function isParticipant() {
          return request.auth != null && 
            exists(/databases/$(database)/documents/privateChats/$(chatId)) &&
            get(/databases/$(database)/documents/privateChats/$(chatId)).data.participants != null &&
            request.auth.uid in get(/databases/$(database)/documents/privateChats/$(chatId)).data.participants;
        }
        
        // Allow participants to read messages
        allow read: if isParticipant();
        
        // Allow participants to create messages
        allow create: if isParticipant() &&
          request.resource.data.userId == request.auth.uid;
        
        // Allow message authors to update their own messages (for editing)
        // Allow any participant to update readBy field (read receipts)
        allow update: if isParticipant() && (
          (resource.data.userId == request.auth.uid && 
           request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'toxicityConfidence', 'toxicityReason', 'toxicityMethod', 'edited', 'editedAt', 'attachment', 'replyTo', 'replyToText', 'replyToUserName', 'mentions'])) ||
          (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy']))
        );
        
        // Allow message authors to delete their own messages
        // Allow admins to delete any message in private chats
        allow delete: if isParticipant() && (
          resource.data.userId == request.auth.uid ||
          isAdmin()
        );
      }
    }
    
    // Typing indicators collection
    match /typing/{userId} {
      // Allow users to read typing status
      allow read: if request.auth != null;
      // Allow users to update their own typing status
      allow create, update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Pinned messages collection (if used as separate collection)
    // Note: Currently pinned messages are stored as a field in messages collection
    // This rule is for future use if a separate collection is needed
    match /pinnedMessages/{messageId} {
      // Allow authenticated users to read pinned messages
      allow read: if request.auth != null;
      // Allow admins to create/update/delete pinned messages
      allow create, update, delete: if isAdmin();
    }
  }
}
