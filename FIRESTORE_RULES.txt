rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is admin
    function isAdmin() {
      return request.auth != null && 
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin1');
    }
    
    // Messages collection
    match /messages/{messageId} {
      // Allow authenticated users to read all messages
      allow read: if request.auth != null;
      
      // Allow authenticated users to create messages
      allow create: if request.auth != null;
      
      // Allow users to update their own messages (for editing)
      // Allow any authenticated user to update readBy field (read receipts)
      // Allow admins to update any field
      allow update: if request.auth != null && (
        (resource.data.userId == request.auth.uid && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'edited', 'editedAt', 'reactions'])) ||
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy'])) ||
        isAdmin()
      );
      
      // Allow message authors and admins to delete messages
      allow delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        isAdmin()
      );
    }
    
    // Users collection
    match /users/{userId} {
      // Allow authenticated users to read any user document
      allow read: if request.auth != null;
      
      // Allow users to create their own user document (during registration)
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Allow users to update their own user document
      // Students can update their profile fields: name, studentEmail, personalEmail, phoneNumber, bio, profilePicture, course, yearOfStudy, dateOfBirth, address, updatedAt, lastSeen, isOnline
      // Admins can update any user document (including other users)
      allow update: if request.auth != null && (
        isAdmin() ||
        (request.auth.uid == userId && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly([
           'name', 
           'studentEmail', 
           'personalEmail', 
           'phoneNumber', 
           'bio', 
           'profilePicture', 
           'course', 
           'yearOfStudy', 
           'dateOfBirth', 
           'address', 
           'updatedAt', 
           'lastSeen', 
           'isOnline'
         ]))
      );
      
      // Allow admins to delete users
      allow delete: if isAdmin();
    }
    
    // Group messages collection
    match /groupMessages/{messageId} {
      // Allow authenticated users to read group messages
      allow read: if request.auth != null;
      
      // Allow authenticated users to create group messages
      allow create: if request.auth != null;
      
      // Allow users to update their own messages (for editing)
      // Allow any authenticated user to update readBy field (read receipts)
      // Allow admins to update any field
      allow update: if request.auth != null && (
        (resource.data.userId == request.auth.uid && 
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'edited', 'editedAt', 'reactions'])) ||
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy'])) ||
        isAdmin()
      );
      
      // Allow message authors and admins to delete messages
      allow delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        isAdmin()
      );
    }
    
    // Reports collection
    match /reports/{reportId} {
      // Allow authenticated users to create reports
      allow create: if request.auth != null;
      
      // Allow admins to read all reports
      allow read: if isAdmin();
      
      // Allow admins to update report status
      allow update: if isAdmin();
      
      // Allow admins to delete reports
      allow delete: if isAdmin();
    }
    
    // Audit logs collection
    match /auditLogs/{logId} {
      // Only admins can read audit logs
      allow read: if isAdmin();
      
      // Only admins can create audit logs (via backend/admin functions)
      allow create: if isAdmin();
      
      // Only admins can update audit logs
      allow update: if isAdmin();
      
      // Only admins can delete audit logs
      allow delete: if isAdmin();
    }
    
    // Groups collection
    match /groups/{groupId} {
      // Allow authenticated users to read groups they are members of
      allow read: if request.auth != null && (
        resource.data.members != null && request.auth.uid in resource.data.members
      );
      
      // Allow authenticated users to create groups
      allow create: if request.auth != null && 
        request.resource.data.createdBy == request.auth.uid &&
        request.resource.data.members != null &&
        request.auth.uid in request.resource.data.members;
      
      // Allow group members to update group (add/remove members, update name/description)
      allow update: if request.auth != null && (
        resource.data.members != null && request.auth.uid in resource.data.members
      );
      
      // Allow group admins to delete groups
      allow delete: if request.auth != null && (
        resource.data.admins != null && request.auth.uid in resource.data.admins
      );
    }
    
    // Group messages collection (subcollection)
    match /groups/{groupId}/groupMessages/{messageId} {
      // Allow group members to read messages
      allow read: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members;
      
      // Allow group members to create messages
      allow create: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members &&
        request.resource.data.userId == request.auth.uid;
      
      // Allow message authors to update their own messages (for editing)
      // Allow any group member to update readBy field (read receipts)
      allow update: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members && (
          (resource.data.userId == request.auth.uid && 
           request.resource.data.diff(resource.data).affectedKeys().hasOnly(['text', 'displayText', 'toxic', 'edited', 'editedAt'])) ||
          (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['readBy']))
        );
      
      // Allow message authors and group admins to delete messages
      allow delete: if request.auth != null && 
        exists(/databases/$(database)/documents/groups/$(groupId)) &&
        get(/databases/$(database)/documents/groups/$(groupId)).data.members != null &&
        request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.members && (
          resource.data.userId == request.auth.uid ||
          (get(/databases/$(database)/documents/groups/$(groupId)).data.admins != null &&
           request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.admins)
        );
    }
  }
}
